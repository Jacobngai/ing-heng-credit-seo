// generate-pages.js (concept)
const locations = require('./data/locations.json');
const brands = require('./data/brands.json');
const useCases = require('./data/use-cases.json');
const equipment = require('./data/equipment-types.json');
const languages = ['en', 'ms', 'zh'];

// Generate all combinations
for (const lang of languages) {
  for (const equip of equipment) {
    // Location pages
    for (const loc of locations) {
      generatePage('location', lang, equip, loc);
    }
    // Brand pages
    for (const brand of brands) {
      generatePage('brand', lang, equip, brand);
    }
    // Use-case pages
    for (const useCase of useCases) {
      generatePage('use-case', lang, equip, useCase);
    }
  }
}
